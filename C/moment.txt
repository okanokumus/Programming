double kutle();
double uzaklik();
int main(){

	double kg,d;	

	kg=kutle();	
	printf("kutle = %.4lf kg\n",kg);
    printf("\n");
	d=uzaklik();	
	printf("uzaklik = %.4lf m\n",d);
    printf("\n");
	printf("yercekimi= %.4lf m/(s^2) \n",g);
	printf("\n");
	printf("kuvvet = kutle * yercekimi = %.4lf Newton kadardir\n",(kg)*g);

getch();

double kutle(){
	char unit;
	double kg,gr;
	printf("kutle birimi: kg icin k, gram icin g ... ");
	scanf(" %c",&unit);
	fflush(stdin);
	switch (unit){
	case 'k':
	printf("kutleyi gir: ");
	scanf(" %lf",&kg);
	fflush(stdin);
	break;
	case 'g':
	printf("kutleyi gir: ");
	scanf(" %lf",&gr);
	fflush(stdin); /* */
	kg=gr/1000;
	break;
	default:
    printf("sadece k veya g \n");
    break;
	}
	return kg;
}

double uzaklik(){
	double cm,mt;	
	char u;
	printf("uzaklik birimi: cm icin c, metre icin m ... ");
	scanf(" %c",&u);
    fflush(stdin);	
	switch (u){
	case 'm':
	printf("uzakligi gir: ");
	scanf(" %lf",&mt);
	fflush(stdin);
	break;
	case 'c':
	printf("uzakligi gir: ");
	scanf(" %lf",&cm);
	
	mt=cm/100;
	break;
	default:
    printf("sadece mt veya c \n");
    break;	
	}
	return mt;
}
